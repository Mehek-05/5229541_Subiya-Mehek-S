#include <stdio.h>

void minimumBribes(int q[], int n) {
    int bribes = 0;
    for (int i = 0; i < n; i++) {
        // Check if current person has moved more than 2 positions ahead
        if (q[i] - (i + 1) > 2) {
            printf("Too chaotic\n");
            return;
        }
    }

    // Count the number of bribes by checking only relevant positions
    for (int i = 0; i < n; i++) {
        // Start from the original position of q[i] minus 2 (or 0 if negative)
        int start = q[i] - 2;
        if (start < 0) start = 0;
        for (int j = start; j < i; j++) {
            if (q[j] > q[i]) {
                bribes++;
            }
        }
    }

    printf("%d\n", bribes);
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        int q[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &q[i]);
        }
        minimumBribes(q, n);
    }
    return 0;
}
