#include <stdio.h>
#include <string.h>

const char* balancedSums(int arr[], int n) {
    long long totalSum = 0;
    long long leftSum = 0;

    for (int i = 0; i < n; i++) {
        totalSum += arr[i];
    }

    for (int i = 0; i < n; i++) {
        // Remove current element from totalSum to get right sum
        totalSum -= arr[i];

        if (leftSum == totalSum) {
            return "YES";
        }

        // Add current element to leftSum for next iteration
        leftSum += arr[i];
    }

    return "NO";
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        int arr[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
        }
        printf("%s\n", balancedSums(arr, n));
    }
    return 0;
}
