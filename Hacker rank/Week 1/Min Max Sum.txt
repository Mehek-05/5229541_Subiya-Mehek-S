#include <stdio.h> 
int main() { 
    long arr[5], sum = 0, min, max; 
 
    for (int i = 0; i < 5; i++) { 
        scanf("%ld", &arr[i]); 
        sum += arr[i]; 
    } 
 
    min = max = arr[0]; 
    for (int i = 1; i < 5; i++) { 
        if (arr[i] < min) min = arr[i]; 
        if (arr[i] > max) max = arr[i]; 
    } 
 
    printf("%ld %ld\n", sum - max, sum - min); 
    return 0; 
}
